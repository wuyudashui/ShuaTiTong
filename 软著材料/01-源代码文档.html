1/** Fisher-Yates shuffle — returns a new shuffled copy */
2export function shuffleArray<T>(arr: T[]): T[] {
3  const a = [...arr];
4  for (let i = a.length - 1; i > 0; i--) {
5    const j = Math.floor(Math.random() * (i + 1));
6    [a[i], a[j]] = [a[j], a[i]];
7  }
8  return a;
